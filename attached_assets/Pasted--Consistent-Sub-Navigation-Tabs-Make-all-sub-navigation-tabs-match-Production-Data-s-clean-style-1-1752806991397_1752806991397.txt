1. UNIFIED SUB-TAB COMPONENT:
Create a reusable sub-tab navigation component:
typescriptinterface SubTab {
  id: string;
  label: string;
  icon: string;
  isActive?: boolean;
  requiresRole?: string;
}

export const SubTabNavigation: React.FC<{
  tabs: SubTab[];
  activeTab: string;
  onTabChange: (tabId: string) => void;
}> = ({ tabs, activeTab, onTabChange }) => {
  const currentUser = getCurrentUser();
  
  const visibleTabs = tabs.filter(tab => 
    !tab.requiresRole || currentUser.role === tab.requiresRole || currentUser.role === 'Corporate'
  );
  
  return (
    <div className="sub-tab-navigation">
      {visibleTabs.map(tab => (
        <button
          key={tab.id}
          className={`sub-tab-btn ${activeTab === tab.id ? 'active' : ''}`}
          onClick={() => onTabChange(tab.id)}
        >
          {tab.icon} {tab.label}
        </button>
      ))}
    </div>
  );
};
