const handleSkipTask = (taskId: number, reason: string) => {
  setTasks(prevTasks => 
    prevTasks.map(task => 
      task.id === taskId 
        ? { 
            ...task, 
            status: 'skipped',
            skipReason: reason,
            skippedAt: new Date().toISOString()
          }
        : task
    )
  );
  setShowTaskModal(false);
  
  // Send alert to managers about skipped task
  console.log(`Task skipped: ${taskId}, Reason: ${reason}`);
};
