1. CREATE THE EDIT/ADD ITEM MODAL:
   Create a modal that handles both editing existing items and adding new ones:
   
   ```jsx
   const InventoryModal = ({ item, onSave, onClose, mode }) => {
     const [formData, setFormData] = useState({
       name: item?.name || '',
       quantity: item?.quantity || 0,
       unit: item?.unit || 'units',
       minQuantity: item?.minQuantity || 0,
       reorderAmount: item?.reorderAmount || 0,
       location: item?.location || '',
       category: item?.category || 'Supplies'
     });
     
     const handleSubmit = (e) => {
       e.preventDefault();
       
       if (!formData.name || !formData.location) {
         alert('Please fill in all required fields');
         return;
       }
       
       const itemData = {
         ...formData,
         quantity: parseInt(formData.quantity),
         minQuantity: parseInt(formData.minQuantity),
         reorderAmount: parseInt(formData.reorderAmount),
         id: item?.id || Date.now()
       };
       
       onSave(itemData);
     };
     
     return (
       <div className="modal-overlay">
         <div className="modal-content">
           <h2>{mode === 'edit' ? 'Edit Inventory Item' : 'Add New Item'}</h2>
           
           <form onSubmit={handleSubmit}>
             {/* Item Name */}
             <label>
               Item Name *
               <input
                 type="text"
                 value={formData.name}
                 onChange={(e) => setFormData({...formData, name: e.target.value})}
                 required
                 placeholder="e.g., Arugula Seeds"
               />
             </label>
             
             {/* Current Quantity */}
             <label>
               Current Quantity *
               <div className="quantity-input-group">
                 <input
                   type="number"
                   value={formData.quantity}
                   onChange={(e) => setFormData({...formData, quantity: e.target.value})}
                   min="0"
                   required
                 />
                 <select
                   value={formData.unit}
                   onChange={(e) => setFormData({...formData, unit: e.target.value})}
                   className="unit-select"
                 >
                   <option value="units">Units</option>
                   <option value="grams">Grams</option>
                   <option value="mL">mL</option>
                   <option value="bags">Bags</option>
                   <option value="lbs">Pounds</option>
                 </select>
               </div>
             </label>
             
             {/* Minimum Quantity */}
             <label>
               Minimum Quantity *
               <input
                 type="number"
                 value={formData.minQuantity}
                 onChange={(e) => setFormData({...formData, minQuantity: e.target.value})}
                 min="0"
                 required
                 title="Quantity below which item should be reordered"
               />
               <span className="field-help">Alert threshold for low stock</span>
             </label>
             
             {/* Reorder Amount */}
             <label>
               Reorder Amount *
               <input
                 type="number"
                 value={formData.reorderAmount}
                 onChange={(e) => setFormData({...formData, reorderAmount: e.target.value})}
                 min="0"
                 required
                 title="Suggested quantity to order when restocking"
               />
               <span className="field-help">Suggested order quantity</span>
             </label>
             
             {/* Location */}
             <label>
               Storage Location *
               <input
                 type="text"
                 value={formData.location}
                 onChange={(e) => setFormData({...formData, location: e.target.value})}
                 required
                 placeholder="e.g., Seed Storage A"
               />
             </label>
             
             {/* Category */}
             <label>
               Category
               <select
                 value={formData.category}
                 onChange={(e) => setFormData({...formData, category: e.target.value})}
               >
                 <option value="Seeds">Seeds</option>
                 <option value="Nutrients">Nutrients</option>
                 <option value="Supplies">Supplies</option>
                 <option value="Equipment">Equipment</option>
                 <option value="Packaging">Packaging</option>
                 <option value="Other">Other</option>
               </select>
             </label>
             
             <div className="modal-actions">
               <button type="submit" className="btn-save">
                 {mode === 'edit' ? 'Save Changes' : 'Add Item'}
               </button>
               <button type="button" onClick={onClose} className="btn-cancel">
                 Cancel
               </button>
             </div>
           </form>
         </div>
       </div>
     );
   };

UPDATE THE INVENTORY PAGE TO INCLUDE MODALS:
Add state and handlers for the modals:
javascriptconst Inventory = () => {
  const [items, setItems] = useState(mockInventoryItems);
  const [showModal, setShowModal] = useState(false);
  const [editingItem, setEditingItem] = useState(null);
  const [modalMode, setModalMode] = useState('add'); // 'add' or 'edit'
  
  // Handler for adding new item
  const handleAddItem = () => {
    setEditingItem(null);
    setModalMode('add');
    setShowModal(true);
  };
  
  // Handler for editing existing item
  const handleEdit = (item) => {
    setEditingItem(item);
    setModalMode('edit');
    setShowModal(true);
  };
  
  // Save handler for both add and edit
  const handleSaveItem = (itemData) => {
    if (modalMode === 'edit') {
      // Update existing item
      setItems(prevItems => 
        prevItems.map(item => 
          item.id === itemData.id ? itemData : item
        )
      );
      console.log('Item updated:', itemData.name);
    } else {
      // Add new item
      setItems(prevItems => [...prevItems, itemData]);
      console.log('New item added:', itemData.name);
    }
    
    setShowModal(false);
    setEditingItem(null);
  };
  
  // Update the page header to show Add Item button for managers
  return (
    <div className="inventory-page">
      <div className="page-header">
        <h1>üì¶ Inventory Management</h1>
        {isManager && (
          <button onClick={handleAddItem} className="btn-primary">
            + Add Item
          </button>
        )}
      </div>
      
      {/* ... rest of the inventory content ... */}
      
      {/* Modal */}
      {showModal && (
        <InventoryModal
          item={editingItem}
          mode={modalMode}
          onSave={handleSaveItem}
          onClose={() => {
            setShowModal(false);
            setEditingItem(null);
          }}
        />
      )}
    </div>
  );
};
