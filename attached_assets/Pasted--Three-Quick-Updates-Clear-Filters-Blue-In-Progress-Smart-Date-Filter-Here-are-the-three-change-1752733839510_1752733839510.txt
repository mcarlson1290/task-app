UPDATE DATE FILTER LOGIC:
Modify the filter function to include overdue tasks when "today" is selected:
javascriptconst filterTasksByDate = (tasks, selectedDate) => {
  if (!selectedDate) return tasks;
  
  const today = new Date().toISOString().split('T')[0];
  const isToday = selectedDate === today;
  
  return tasks.filter(task => {
    // If today is selected, show today's tasks AND overdue tasks
    if (isToday) {
      return task.completionDate === selectedDate || task.isOverdue;
    }
    
    // For other dates, only show tasks due on that exact date
    return task.completionDate === selectedDate;
  });
};
