jsxconst StaffEditView = ({ staff, searchTerm, setSearchTerm, sortBy, setSortBy, onEdit, isManager }) => {
  // Filter and sort staff
  const filteredStaff = staff.filter(person =>
    person.fullName.toLowerCase().includes(searchTerm.toLowerCase()) ||
    person.email.toLowerCase().includes(searchTerm.toLowerCase()) ||
    person.rolesAssigned.some(role => role.toLowerCase().includes(searchTerm.toLowerCase()))
  );
  
  const sortedStaff = [...filteredStaff].sort((a, b) => {
    switch (sortBy) {
      case 'name':
        return a.fullName.localeCompare(b.fullName);
      case 'dateHired':
        return new Date(b.dateHired) - new Date(a.dateHired);
      case 'payRate':
        return b.payRate - a.payRate;
      case 'tasks':
        return b.tasksCompleted - a.tasksCompleted;
      default:
        return 0;
    }
  });
  
  return (
    <div className="staff-edit-view">
      {/* Toolbar */}
      <div className="staff-toolbar">
        <input
          type="text"
          placeholder="üîç Search by name, email, or role..."
          value={searchTerm}
          onChange={(e) => setSearchTerm(e.target.value)}
          className="search-input"
        />
        
        <select value={sortBy} onChange={(e) => setSortBy(e.target.value)}>
          <option value="name">Sort by Name</option>
          <option value="dateHired">Sort by Hire Date</option>
          <option value="payRate">Sort by Pay Rate</option>
          <option value="tasks">Sort by Tasks Completed</option>
        </select>

CREATE THE STAFF ANALYTICS VIEW:
jsxconst StaffAnalyticsView = ({ staff, tasks }) => {
  // Calculate analytics
  const activeStaff = staff.filter(s => s.activeStatus === 'active');
  const totalTasks = tasks.length;
  const avgTasksPerPerson = totalTasks / activeStaff.length;
  
  // Calculate labor costs
  const calculateLaborCosts = () => {
    return tasks.reduce((total, task) => {
      if (task.status === 'completed' && task.actualDuration) {
        const assignedStaff = staff.find(s => s.fullName === task.assignedTo);
        if (assignedStaff) {
          const hours = parseDurationToHours(task.actualDuration);
          return total + (hours * assignedStaff.payRate);
        }
      }
      return total;
    }, 0);
  };
  
  return (
    <div className="staff-analytics-view">
      {/* Summary Cards */}
      <div className="analytics-cards">
        <div className="analytics-card">
          <h3>Total Staff</h3>
          <div className="metric">{staff.length}</div>
          <div className="sub-metric">{activeStaff.length} active</div>
        </div>
        
        <div className="analytics-card">
          <h3>Avg Tasks/Day</h3>
          <div className="metric">{avgTasksPerPerson.toFixed(1)}</div>
          <div className="sub-metric">per person</div>
        </div>
        
        <div className="analytics-card">
          <h3>On-Time Rate</h3>
          <div className="metric">
            {(staff.reduce((sum, s) => sum + s.onTimeRate, 0) / staff.length).toFixed(1)}%
          </div>
          <div className="sub-metric">average</div>
        </div>
        
        <div className="analytics-card">
          <h3>Labor Cost (30d)</h3>
          <div className="metric">${calculateLaborCosts().toFixed(2)}</div>
          <div className="sub-metric">based on completed tasks</div>
        </div>
      </div>
      
      {/* Performance Chart */}
      <div className="performance-section">
        <h3>Individual Performance Metrics</h3>
        <div className="performance-grid">
          {activeStaff.map(person => (
            <StaffPerformanceCard 
              key={person.id}
              staff={person}
              tasks={tasks.filter(t => t.assignedTo === person.fullName)}
            />
          ))}
        </div>
      </div>
      
      {/* Training Overview */}
      <div className="training-overview">
        <h3>Training Completion Status</h3>
        <div className="training-matrix">
          {renderTrainingMatrix(staff)}
        </div>
      </div>
    </div>
  );
};

CREATE PERFORMANCE CARD COMPONENT:
jsxconst StaffPerformanceCard = ({ staff, tasks }) => {
  const completedTasks = tasks.filter(t => t.status === 'completed');
  const last7Days = tasks.filter(t => {
    const taskDate = new Date(t.completedAt || t.createdAt);
    const weekAgo = new Date();
    weekAgo.setDate(weekAgo.getDate() - 7);
    return taskDate > weekAgo;
  });
  
  // Calculate labor cost for this person
  const laborCost = completedTasks.reduce((total, task) => {
    if (task.actualDuration) {
      const hours = parseDurationToHours(task.actualDuration);
      return total + (hours * staff.payRate);
    }
    return total;
  }, 0);
  
  return (
    <div className="performance-card">
      <h4>{staff.fullName}</h4>
      
      <div className="performance-metrics">
        <div className="metric-row">
          <span>Tasks (7d):</span>
          <span>{last7Days.length}</span>
        </div>
        <div className="metric-row">
          <span>Avg Duration:</span>
          <span>{staff.avgTaskDuration}</span>
        </div>
        <div className="metric-row">
          <span>On-Time Rate:</span>
          <span className={staff.onTimeRate > 95 ? 'good' : 'warning'}>
            {staff.onTimeRate}%
          </span>
        </div>
        <div className="metric-row">
          <span>Labor Cost (30d):</span>
          <span>${laborCost.toFixed(2)}</span>
        </div>
      </div>
      
      <div className="task-breakdown">
        <h5>Tasks by Type</h5>
        {renderTaskTypeBreakdown(tasks)}
      </div>
    </div>
  );
};
