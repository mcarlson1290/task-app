CREATE THE COURSE CARD COMPONENT:
jsxconst CourseCard = ({ course, onStart, onResume }) => {
  const getStatusColor = () => {
    switch (course.status) {
      case 'completed': return 'status-completed';
      case 'in-progress': return 'status-progress';
      default: return 'status-not-started';
    }
  };
  
  const getActionButton = () => {
    switch (course.status) {
      case 'completed':
        return (
          <button className="btn-completed" disabled>
            ‚úÖ Completed
          </button>
        );
      case 'in-progress':
        return (
          <button onClick={() => onResume(course)} className="btn-resume">
            ‚ñ∂Ô∏è Resume Course
          </button>
        );
      default:
        return (
          <button onClick={() => onStart(course)} className="btn-start">
            Start Course
          </button>
        );
    }
  };
  
  return (
    <div className={`course-card ${getStatusColor()}`}>
      <div className="course-icon">{course.icon}</div>
      
      <div className="course-header">
        <h3>{course.title}</h3>
        {course.requiresApproval && (
          <span className="approval-badge">Manager Approval Required</span>
        )}
      </div>
      
      <p className="course-description">{course.description}</p>
      
      <div className="course-meta">
        <span>üìë {course.sections} sections</span>
        <span>‚è±Ô∏è {course.estimatedTime}</span>
      </div>
      
      {course.status === 'in-progress' && (
        <div className="progress-section">
          <div className="progress-text">
            Progress: {course.progress}/{course.sections} sections
          </div>
          <div className="progress-bar">
            <div 
              className="progress-fill"
              style={{ width: `${(course.progress / course.sections) * 100}%` }}
            />
          </div>
        </div>
      )}
      
      {course.roleAwarded && (
        <div className="role-award">
          üèÖ Awards: <strong>{course.roleAwarded}</strong> role
        </div>
      )}
      
      <div className="course-actions">
        {getActionButton()}
        {course.status === 'completed' && course.completedDate && (
          <span className="completed-date">
            Completed {new Date(course.completedDate).toLocaleDateString()}
          </span>
        )}
      </div>
    </div>
  );
};

CREATE A SIMPLE COURSE MODAL:
jsxconst CourseModal = ({ course, onClose, onComplete }) => {
  const [currentSection, setCurrentSection] = useState(course.progress || 0);
  const [sectionCompleted, setSectionCompleted] = useState(false);
  
  const totalSections = course.sections;
  const isLastSection = currentSection === totalSections - 1;
  
  const handleNext = () => {
    if (isLastSection && sectionCompleted) {
      // Course complete!
      onComplete(course.id);
    } else if (sectionCompleted) {
      // Move to next section
      setCurrentSection(currentSection + 1);
      setSectionCompleted(false);
    }
  };
  
  const handleSectionComplete = () => {
    setSectionCompleted(true);
  };
  
  return (
    <div className="modal-overlay">
      <div className="modal-content course-modal">
        <div className="modal-header">
          <h2>{course.title}</h2>
          <button onClick={onClose} className="close-btn">‚úï</button>
        </div>
        
        <div className="section-progress">
          Section {currentSection + 1} of {totalSections}
        </div>
        
        <div className="section-content">
          {/* Placeholder content for now */}
          <div className="placeholder-lesson">
            <h3>Section {currentSection + 1}: {getSectionTitle(course.id, currentSection)}</h3>
            <div className="lesson-content">
              <p>This is where the actual course content would go:</p>
              <ul>
                <li>üìπ Video tutorials</li>
                <li>üìÑ Text instructions</li>
                <li>üñºÔ∏è Diagrams and images</li>
                <li>‚úÖ Interactive checklists</li>
                <li>‚ùì Quiz questions</li>
              </ul>
              
              <div className="demo-complete">
                <label>
                  <input
                    type="checkbox"

                    onChange={(e) => setSectionCompleted(e.target.checked)}
                  />
                  I have reviewed this section
                </label>
              </div>
            </div>
          </div>
        </div>
        
        <div className="modal-footer">
          <div className="progress-indicator">
            {[...Array(totalSections)].map((_, i) => (
              <span
                key={i}
                className={`progress-dot ${i <= currentSection ? 'completed' : ''}`}
              />
            ))}
          </div>
          
          <button
            onClick={handleNext}
            disabled={!sectionCompleted}
            className={`btn-next ${isLastSection ? 'btn-complete' : ''}`}
          >
            {isLastSection ? 'üéâ Complete Course' : 'Next Section ‚Üí'}
          </button>
        </div>
      </div>
    </div>
  );
};

javascriptconst handleCourseComplete = (courseId) => {
  // Update course status
  setCourses(prevCourses =>
    prevCourses.map(course =>
      course.id === courseId
        ? {
            ...course,
            status: 'completed',
            progress: course.sections,
            completedDate: new Date().toISOString()
          }
        : course
    )
  );
  
  // Get the completed course
  const completedCourse = courses.find(c => c.id === courseId);
  
  if (completedCourse) {
