        default:
          return true;
      }
    });
    
    return coursePrereqsMet && requirementsMet;

CREATE REQUIREMENT MODAL:
jsxconst RequirementModal = ({ onAdd, onClose }) => {
  const [type, setType] = useState('age');
  const [value, setValue] = useState('');
  const [customLabel, setCustomLabel] = useState('');
  
  const requirementTypes = {
    age: { label: 'Minimum Age', unit: 'years' },
    tenure: { label: 'Employment Duration', unit: 'days' },
    license: { label: 'License/Certification', options: ['drivers', 'forklift', 'hazmat'] },
    certification: { label: 'Special Certification', custom: true },
    approval: { label: 'Manager Approval', noValue: true },
    training: { label: 'External Training', custom: true }
  };
  
  const handleAdd = () => {
    let label = '';
    
    switch (type) {
      case 'age':
        label = `Must be ${value} or older`;
        break;
      case 'tenure':
        label = `Employed for at least ${value} days`;
        break;
      case 'license':
        label = `Valid ${value} license required`;
        break;
      case 'approval':
        label = 'Requires manager approval';
        break;
      default:
        label = customLabel || `${requirementTypes[type].label} required`;
    }
    
    onAdd({ type, value, label });
  };
  
  return (
    <div className="modal-overlay">
      <div className="modal-content small-modal">
        <h3>Add Requirement</h3>
        
        <label>
          Requirement Type
          <select value={type} onChange={(e) => setType(e.target.value)}>
            {Object.entries(requirementTypes).map(([key, config]) => (
              <option key={key} value={key}>{config.label}</option>
            ))}
          </select>
        </label>
        
        {!requirementTypes[type].noValue && (
          <label>
            {requirementTypes[type].custom ? 'Description' : 'Value'}
            {requirementTypes[type].options ? (
              <select value={value} onChange={(e) => setValue(e.target.value)}>
                <option value="">Select...</option>
                {requirementTypes[type].options.map(opt => (
                  <option key={opt} value={opt}>{opt}</option>
                ))}
              </select>
            ) : (
              <input
                type={requirementTypes[type].custom ? 'text' : 'number'}
                value={value}
                onChange={(e) => setValue(e.target.value)}
                placeholder={requirementTypes[type].unit || 'Enter value'}
              />
            )}
          </label>
        )}
        
        {requirementTypes[type].custom && (
          <label>
            Display Label
            <input
              type="text"
              value={customLabel}
              onChange={(e) => setCustomLabel(e.target.value)}
              placeholder="e.g., First Aid Certification required"
            />
          </label>
        )}
        
        <div className="modal-actions">
          <button onClick={handleAdd} className="btn-primary">
            Add
          </button>
          <button onClick={onClose} className="btn-cancel">
            Cancel
          </button>
        </div>
      </div>
    </div>
  );
};
