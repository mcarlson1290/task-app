OVERDUE LOGIC:
Tasks become overdue at 8:27 PM on their due date (per the requirements doc):
typescriptconst checkIfOverdue = (dueDate: string) => {
  const now = new Date();
  const due = new Date(dueDate);
  due.setHours(20, 27, 0); // 8:27 PM
  return now > due;
};
