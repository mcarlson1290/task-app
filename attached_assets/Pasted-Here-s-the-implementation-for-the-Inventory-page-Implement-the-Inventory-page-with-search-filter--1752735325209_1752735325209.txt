CREATE THE INVENTORY CARD COMPONENT:
jsxconst InventoryCard = ({ item, isManager, onReorder, onEdit }) => {
  const isLowStock = item.quantity <= item.minQuantity;
  const stockPercentage = (item.quantity / item.minQuantity) * 100;
  
  return (
    <div className={`inventory-card ${isLowStock ? 'low-stock' : ''}`}>
      <div className="card-header">
        <h3>{item.name}</h3>
        <span className={`stock-badge ${isLowStock ? 'badge-low' : 'badge-ok'}`}>
          {isLowStock ? '‚ö†Ô∏è Low Stock' : '‚úÖ In Stock'}
        </span>
      </div>
      
      <div className="card-body">
        <div className="quantity-display">
          <div className="quantity-main">
            <span className="quantity-number">{item.quantity}</span>
            <span className="quantity-unit">{item.unit}</span>
          </div>
          <div className="quantity-min">
            Min: {item.minQuantity} {item.unit}
          </div>
        </div>
        
        <div className="stock-bar">
          <div 
            className="stock-fill"
            style={{
              width: `${Math.min(stockPercentage, 100)}%`,
              backgroundColor: stockPercentage > 50 ? '#10b981' : '#ef4444'
            }}
          />
        </div>
        
        <div className="item-details">
          <p>üìç Location: {item.location}</p>
          <p>üì¶ Reorder Amount: {item.reorderAmount} {item.unit}</p>
        </div>
      </div>
      
      <div className="card-actions">
        <button 
          onClick={() => onReorder(item)}
          className="btn-reorder"
        >
          üìß Reorder
        </button>
        {isManager && (
          <button 
            onClick={() => onEdit(item)}
            className="btn-edit"
          >
            ‚úèÔ∏è Edit
          </button>
        )}
      </div>
    </div>
  );
};

ADD REORDER FUNCTIONALITY:
javascriptconst handleReorder = (item) => {
  const subject = `Reorder Request: ${item.name}`;
  const body = `
    Item: ${item.name}
    Current Quantity: ${item.quantity} ${item.unit}
    Minimum Quantity: ${item.minQuantity} ${item.unit}
    Requested Reorder Amount: ${item.reorderAmount} ${item.unit}
    Location: ${item.location}
    
    Requesting Location: Grow Space
    Requested By: ${currentUser.name}
    Date: ${new Date().toLocaleDateString()}
  `;
  
  // Create mailto link
  const mailtoLink = `mailto:accounting@growspace.farm?subject=${encodeURIComponent(subject)}&body=${encodeURIComponent(body)}`;
  window.location.href = mailtoLink;
  
  // Show confirmation
  alert(`Reorder email opened for ${item.name}`);
};
