typescriptconst filterTasksByDate = (tasks: Task[], selectedDate: Date) => {
  return tasks.filter(task => {
    if (typeof task.completionDate === 'string') {
      return isSameDay(new Date(task.completionDate), selectedDate);
    } else {
      // For date ranges, check if selected date is within range
      const start = new Date(task.completionDate.start);
      const end = new Date(task.completionDate.end);
      return isWithinInterval(selectedDate, { start, end });
    }
  });
};
