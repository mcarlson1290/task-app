Fix 2: Checklist Loading on First Click
javascript// Fix the timing issue with checklist loading
const TaskModal = ({ task, onClose }) => {
  const [checklist, setChecklist] = useState([]);
  const [isLoading, setIsLoading] = useState(true);
  
  useEffect(() => {
    // Immediate load without delays
    if (task && task.id) {
      setIsLoading(true);
      
      // Load checklist directly
      const loadChecklist = async () => {
        try {
          // If checklist is part of task object
          if (task.checklist) {
            setChecklist([...task.checklist]);
          } else {
            // If needs to be loaded separately
            const taskData = await loadTaskDetails(task.id);
            setChecklist(taskData.checklist || []);
          }
        } catch (error) {
          console.error('Error loading checklist:', error);
          setChecklist([]);
        } finally {
          setIsLoading(false);
        }
      };
      
      loadChecklist();
    }
  }, [task?.id]);
  
  return (
    <div className="task-modal">
      {/* Header */}
      <div className="modal-header">
        <h2>{task?.title || 'Loading...'}</h2>
        <button onClick={onClose} className="close-button">×</button>
      </div>
      
      {/* Checklist Section */}
      <div className="checklist-section">
        <h3>Checklist</h3>
        {isLoading ? (
          <div className="loading">Loading checklist...</div>
        ) : checklist.length > 0 ? (
          <div className="checklist-items">
            {checklist.map((item, index) => (
              <div key={index} className="checklist-item">
                <input
                  type="checkbox"
                  id={`check-${index}`}
                  className="checklist-checkbox"
                />
                <label 
                  htmlFor={`check-${index}`}
                  className="checklist-item-text"
                >
                  {item.text || item}
                </label>
              </div>
            ))}
          </div>
        ) : (
          <p>No checklist items</p>
        )}
      </div>
      
      {/* Action Buttons - Stack on mobile */}
      <div className="button-group">
        <button className="btn-complete">Complete Task</button>
        <button className="btn-pause">Pause Task</button>
        <button className="btn-skip">Skip Task</button>
      </div>
    </div>
  );
};

Fix 4: Mobile-Specific Modal Container
javascript// Responsive modal container
const Modal = ({ children, isOpen, onClose }) => {
  if (!isOpen) return null;
  
  return (
    <div className="modal-overlay" onClick={onClose}>
      <div 
        className="modal-container"
        onClick={(e) => e.stopPropagation()}
        style={{
          position: 'fixed',
          top: '50%',
          left: '50%',
          transform: 'translate(-50%, -50%)',
          width: 'calc(100vw - 20px)', // Full width minus margins
          maxWidth: '600px',
          maxHeight: '90vh',
          overflow: 'auto',
          WebkitOverflowScrolling: 'touch', // Smooth iOS scrolling
        }}
      >
        {children}
      </div>
    </div>
  );
};
