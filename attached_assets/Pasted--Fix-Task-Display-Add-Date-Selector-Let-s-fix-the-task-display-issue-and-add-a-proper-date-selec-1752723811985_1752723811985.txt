typescriptconst filterTasks = () => {
  let filtered = [...tasks];
  
  // Filter by category
  if (selectedCategory && selectedCategory !== 'all') {
    filtered = filtered.filter(task => task.type === selectedCategory);
  }
  
  // Filter by date
  if (selectedDate) {
    filtered = filtered.filter(task => {
      if (!task.completionDate) return false;
      
      if (typeof task.completionDate === 'string') {
        return task.completionDate === selectedDate;
      } else {
        // Handle date ranges
        const selected = new Date(selectedDate);
        const start = new Date(task.completionDate.start);
        const end = new Date(task.completionDate.end);
        return selected >= start && selected <= end;
      }
    });
  }
  
  // Filter by status
  if (selectedStatus && selectedStatus !== 'all') {
    filtered = filtered.filter(task => task.status === selectedStatus);
  }
  
  return filtered;
};
