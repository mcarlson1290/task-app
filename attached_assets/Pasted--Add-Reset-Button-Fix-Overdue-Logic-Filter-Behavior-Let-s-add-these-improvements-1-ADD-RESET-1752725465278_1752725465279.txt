ALWAYS SHOW OVERDUE TASKS REGARDLESS OF DATE FILTER:
Update the filter logic to always include overdue tasks:
typescriptconst filterTasks = () => {
  let filtered = [...tasks];
  
  // First, separate overdue tasks
  const overdueTasks = tasks.filter(task => task.isOverdue);
  const nonOverdueTasks = tasks.filter(task => !task.isOverdue);
  
  // Apply filters to non-overdue tasks
  let filteredNonOverdue = nonOverdueTasks;
  
  // Category filter
  if (selectedCategory && selectedCategory !== 'all') {
    filteredNonOverdue = filteredNonOverdue.filter(task => 
      task.type === selectedCategory
    );
  }
  
  // Date filter (only for non-overdue tasks)
  if (selectedDate) {
    filteredNonOverdue = filteredNonOverdue.filter(task => 
      task.completionDate === selectedDate
    );
  }
  
  // Status filter
  if (selectedStatus && selectedStatus !== 'all') {
    filteredNonOverdue = filteredNonOverdue.filter(task => 
      task.status === selectedStatus
    );
    // Also filter overdue tasks by status
    overdueTasks = overdueTasks.filter(task => 
      task.status === selectedStatus
    );
  }
  
  // Priority filter
  if (selectedPriority && selectedPriority !== 'all') {
    filteredNonOverdue = filteredNonOverdue.filter(task => 
      task.priority === selectedPriority
    );
    overdueTasks = overdueTasks.filter(task => 
      task.priority === selectedPriority
    );
  }
  
  // Combine overdue tasks (always shown) with filtered non-overdue tasks
  return [...overdueTasks, ...filteredNonOverdue];
};

UPDATE OVERDUE CALCULATION:
typescriptconst calculateOverdueDays = (dueDate: string) => {
  const now = new Date();
  const due = new Date(dueDate);
  due.setHours(20, 27, 0, 0); // Set to 8:27 PM on due date
  
  const diffTime = now.getTime() - due.getTime();
  const diffDays = Math.floor(diffTime / (1000 * 60 * 60 * 24));
  
  if (diffDays === 0) {
    // Less than 24 hours overdue
    const diffHours = Math.floor(diffTime / (1000 * 60 * 60));
    return `${diffHours} hours`;
  }
  
  return `${diffDays} day${diffDays > 1 ? 's' : ''}`;
};
