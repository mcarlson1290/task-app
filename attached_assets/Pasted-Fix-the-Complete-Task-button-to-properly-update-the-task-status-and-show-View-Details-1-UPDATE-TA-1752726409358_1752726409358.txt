ENSURE handleCompleteTask UPDATES THE TASK:
typescriptconst handleCompleteTask = (taskId: number, notes: string) => {
  console.log('Completing task:', taskId); // Debug log
  
  setTasks(prevTasks => {
    const updatedTasks = prevTasks.map(task => 
      task.id === taskId 
        ? { 
            ...task, 
            status: 'completed', // THIS IS THE KEY CHANGE
            completedAt: new Date().toISOString(),
            completionNotes: notes,
            progress: 100
          }
        : task
    );
    console.log('Updated tasks:', updatedTasks); // Debug log
    return updatedTasks;
  });
  
  // Close modal
  setShowTaskModal(false);
  setSelectedTask(null);
  
  // Optional: Show success feedback
  // showConfetti();
};

typescriptconst handleViewDetails = (taskId: number) => {
  const task = tasks.find(t => t.id === taskId);
  if (task) {
    setSelectedTask(task);
    setShowTaskModal(true);
