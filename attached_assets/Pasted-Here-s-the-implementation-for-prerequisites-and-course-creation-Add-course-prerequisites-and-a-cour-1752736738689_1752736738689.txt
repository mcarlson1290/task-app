UPDATE COURSE CARD TO SHOW PREREQUISITES:
jsxconst CourseCard = ({ course, onStart, onResume, allCourses }) => {
  // Check if prerequisites are met
  const prerequisitesMet = course.prerequisites.every(prereqId => {
    const prereqCourse = allCourses.find(c => c.id === prereqId);
    return prereqCourse?.status === 'completed';
  });
  
  const isLocked = !prerequisitesMet || course.status === 'locked';
  
  const getActionButton = () => {
    if (isLocked) {
      return (
        <button className="btn-locked" disabled>
          üîí Locked
        </button>
      );
    }
    
    switch (course.status) {
      case 'completed':
        return (
          <button className="btn-completed" disabled>
            ‚úÖ Completed
          </button>
        );
      case 'in-progress':
        return (
          <button onClick={() => onResume(course)} className="btn-resume">
            ‚ñ∂Ô∏è Resume Course
          </button>
        );
      default:
        return (
          <button onClick={() => onStart(course)} className="btn-start">
            Start Course
          </button>
        );
    }
  };
  
  const getPrerequisiteNames = () => {
    return course.prerequisites.map(prereqId => {
      const prereq = allCourses.find(c => c.id === prereqId);
      return prereq?.title || 'Unknown';
    });
  };
  
  return (
    <div className={`course-card ${getStatusColor()} ${isLocked ? 'locked' : ''}`}>
      {/* ... existing content ... */}
      
      {course.prerequisites.length > 0 && (
        <div className="prerequisites-section">
          <div className="prereq-label">
            {prerequisitesMet ? '‚úÖ Prerequisites Met' : 'üîí Prerequisites Required:'}
          </div>
          {!prerequisitesMet && (
            <ul className="prereq-list">
              {getPrerequisiteNames().map((name, idx) => (
                <li key={idx}>{name}</li>
              ))}
            </ul>
          )}
        </div>
      )}
      
      {/* ... rest of card ... */}
    </div>
  );
};

CREATE THE COURSE CREATION MODAL:
jsxconst CourseCreationModal = ({ allCourses, onSave, onClose }) => {
  const [courseData, setCourseData] = useState({
    title: '',
    description: '',
    roleAwarded: '',
    sections: 1,
    estimatedTime: '',
    icon: 'üìö',
    prerequisites: [],
    requiresApproval: false
  });
  
  const [sections, setSections] = useState([{ 
    title: '', 
    type: 'text', 
    content: '' 
  }]);
  
  const availableIcons = ['üìö', 'üå±', 'üåæ', 'üßπ', 'üîß', 'ü¶∫', 'üìä', 'üåø', 'üëî'];
  
  const handleAddSection = () => {
    setSections([...sections, { title: '', type: 'text', content: '' }]);
    setCourseData({ ...courseData, sections: sections.length + 1 });
  };
  
  const handleRemoveSection = (index) => {
    const newSections = sections.filter((_, i) => i !== index);
    setSections(newSections);
    setCourseData({ ...courseData, sections: newSections.length });
  };
  
  const handleSave = () => {
    if (!courseData.title || !courseData.roleAwarded) {
      alert('Please fill in required fields');
      return;
    }
    
    const newCourse = {
      ...courseData,
      id: Date.now(),
      status: 'not-started',
      progress: 0,
      sections: sections.length,
      createdBy: getCurrentUser().name,
      createdAt: new Date().toISOString()
    };
    
    onSave(newCourse);
  };
  
  return (
    <div className="modal-overlay">
      <div className="modal-content large-modal">
        <h2>Create New Course</h2>
        
        <div className="form-grid">
          {/* Course Title */}
          <label className="full-width">
            Course Title *
            <input
              type="text"
              value={courseData.title}
              onChange={(e) => setCourseData({...courseData, title: e.target.value})}
              placeholder="e.g., Advanced Hydroponics Training"
            />
          </label>
          
          {/* Description */}
          <label className="full-width">
            Description *
            <textarea
              value={courseData.description}
              onChange={(e) => setCourseData({...courseData, description: e.target.value})}
              placeholder="Brief description of what students will learn..."
              rows={3}
            />
          </label>
          
          {/* Role Awarded */}
          <label>
            Role Awarded *
            <input
              type="text"
              value={courseData.roleAwarded}
              onChange={(e) => setCourseData({...courseData, roleAwarded: e.target.value})}
              placeholder="e.g., Senior Technician"
            />
          </label>
          
          {/* Estimated Time */}
          <label>
            Estimated Time *
            <input
              type="text"
              value={courseData.estimatedTime}
              onChange={(e) => setCourseData({...courseData, estimatedTime: e.target.value})}
              placeholder="e.g., 2 hours"
            />
          </label>
          
          {/* Icon Selection */}
          <label>
            Course Icon
            <div className="icon-selector">
              {availableIcons.map(icon => (
                <button
                  key={icon}
                  type="button"
                  className={`icon-option ${courseData.icon === icon ? 'selected' : ''}`}
                  onClick={() => setCourseData({...courseData, icon})}
                >
                  {icon}
                </button>
              ))}
            </div>
          </label>
          
          {/* Prerequisites */}
          <label>
            Prerequisites
            <div className="prerequisites-selector">
              {allCourses.map(course => (
                <label key={course.id} className="checkbox-label">
                  <input
                    type="checkbox"
                    checked={courseData.prerequisites.includes(course.id)}
                    onChange={(e) => {
                      if (e.target.checked) {
                        setCourseData({
                          ...courseData,
                          prerequisites: [...courseData.prerequisites, course.id]
                        });
                      } else {
                        setCourseData({
                          ...courseData,
                          prerequisites: courseData.prerequisites.filter(id => id !== course.id)
                        });
                      }
                    }}
                  />
                  {course.title}
                </label>
              ))}
            </div>
          </label>
          
          {/* Manager Approval */}
          <label className="checkbox-label">
            <input
              type="checkbox"
              checked={courseData.requiresApproval}
              onChange={(e) => setCourseData({...courseData, requiresApproval: e.target.checked})}
            />
            Requires Manager Approval
          </label>
        </div>
        
        {/* Sections */}
        <div className="sections-builder">
          <h3>Course Sections</h3>
          {sections.map((section, index) => (
            <div key={index} className="section-builder">
              <div className="section-header">
                <h4>Section {index + 1}</h4>
                {sections.length > 1 && (
                  <button
                    type="button"
                    onClick={() => handleRemoveSection(index)}
                    className="btn-remove"
                  >
                    ‚úï
                  </button>
                )}
              </div>
              
              <input
                type="text"
                placeholder="Section title"
                value={section.title}
                onChange={(e) => {
                  const newSections = [...sections];
                  newSections[index].title = e.target.value;
                  setSections(newSections);
                }}
              />
              
              <select
                value={section.type}
                onChange={(e) => {
                  const newSections = [...sections];
                  newSections[index].type = e.target.value;
                  setSections(newSections);
                }}
              >
                <option value="text">Text Content</option>
                <option value="video">Video</option>
                <option value="quiz">Quiz</option>
                <option value="checklist">Checklist</option>
              </select>
              
              <textarea
                placeholder="Section content/instructions..."
                value={section.content}
                onChange={(e) => {
                  const newSections = [...sections];
                  newSections[index].content = e.target.value;
                  setSections(newSections);
                }}
                rows={3}
              />
            </div>
          ))}
          
          <button
            type="button"
            onClick={handleAddSection}
            className="btn-add-section"
          >
            + Add Section
          </button>
        </div>
        
        <div className="modal-actions">
          <button onClick={handleSave} className="btn-save">
            Create Course
          </button>
          <button onClick={onClose} className="btn-cancel">
            Cancel
          </button>
        </div>
      </div>
    </div>
  );
};
