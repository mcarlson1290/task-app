ADD HELPER FUNCTIONS FOR VIDEO EMBEDS:
javascript// Helper functions for video URLs
const isYouTubeUrl = (url) => {
  return url.includes('youtube.com/watch') || url.includes('youtu.be/');
};

const getYouTubeEmbedUrl = (url) => {
  let videoId = '';
  if (url.includes('youtube.com/watch')) {
    videoId = url.split('v=')[1]?.split('&')[0];
  } else if (url.includes('youtu.be/')) {
    videoId = url.split('youtu.be/')[1]?.split('?')[0];
  }
  return `https://www.youtube.com/embed/${videoId}`;
};

const isVimeoUrl = (url) => {
  return url.includes('vimeo.com/');
};

const getVimeoEmbedUrl = (url) => {
  const videoId = url.split('vimeo.com/')[1]?.split('?')[0];
  return `https://player.vimeo.com/video/${videoId}`;
};

// Render rich text content with embedded media
const renderRichContent = (content) => {
  if (!content) return null;
  
  // Split content by media tags
  const parts = content.split(/(\[\[IMAGE:.*?\]\]|\[\[VIDEO:.*?\]\])/);
  
  return parts.map((part, index) => {
    if (part.startsWith('[[IMAGE:')) {
      const url = part.match(/\[\[IMAGE:(.*?)\]\]/)?.[1];
      return url ? (
        <img 
          key={index} 
          src={url} 
          alt="Embedded" 
          className="embedded-image"
          onError={(e) => {
            e.target.src = 'https://via.placeholder.com/400x300?text=Image+Not+Found';
          }}
        />
      ) : null;
    } else if (part.startsWith('[[VIDEO:')) {
      const url = part.match(/\[\[VIDEO:(.*?)\]\]/)?.[1];
      if (!url) return null;
      
      if (isYouTubeUrl(url)) {
        return (
          <iframe
            key={index}
            src={getYouTubeEmbedUrl(url)}
            className="embedded-video"
            frameBorder="0"
            allowFullScreen
          />
        );
      } else if (isVimeoUrl(url)) {
        return (
          <iframe
            key={index}
            src={getVimeoEmbedUrl(url)}
            className="embedded-video"
            frameBorder="0"
            allowFullScreen
          />
        );
      } else {
        return (
          <video key={index} controls className="embedded-video">
            <source src={url} />
          </video>
        );
      }
    } else {
      return <p key={index}>{part}</p>;
    }
  });
};

UPDATE THE COURSE VIEWER TO RENDER MEDIA:
jsx// In the CourseModal component, update section display
const renderSectionForViewing = (section) => {
  switch (section.type) {
    case 'video':
      if (isYouTubeUrl(section.content)) {
        return (
          <iframe
            src={getYouTubeEmbedUrl(section.content)}
            width="100%"
            height="400"
            frameBorder="0"
            allowFullScreen
          />
        );
      } else if (isVimeoUrl(section.content)) {
        return (
          <iframe
            src={getVimeoEmbedUrl(section.content)}
            width="100%"
            height="400"
            frameBorder="0"
            allowFullScreen
          />
        );
      } else {
        return (
          <video width="100%" controls>
            <source src={section.content} />
            Your browser does not support video.
          </video>
        );
      }
      
    case 'image':
      return (
        <div className="image-display">
          <img src={section.content} alt={section.title} />
          {section.caption && <p className="image-caption">{section.caption}</p>}
        </div>
      );
      
    case 'rich-text':
      return (
        <div className="rich-text-display">
          {renderRichContent(section.content)}
        </div>
      );
      
    case 'quiz':
      return (
        <div className="quiz-display">
          <p className="question">{section.question}</p>
          <div className="quiz-options">
            {['A', 'B', 'C', 'D'].map(option => (
              <label key={option} className="quiz-option">
                <input
                  type="radio"
                  name={`quiz-${section.id}`}
                  value={option}
                />
                {section[`option${option}`]}
              </label>
            ))}
          </div>
        </div>
      );
      
    case 'checklist':
      const items = section.content.split('\n').filter(item => item.trim());
      return (
        <div className="checklist-display">
          {items.map((item, idx) => (
            <label key={idx} className="checklist-item">
              <input type="checkbox" />
              {item.replace(/^‚òê\s*/, '')}
            </label>
          ))}
        </div>
      );
      
    default:
      return <div className="text-display">{section.content}</div>;
  }
};
