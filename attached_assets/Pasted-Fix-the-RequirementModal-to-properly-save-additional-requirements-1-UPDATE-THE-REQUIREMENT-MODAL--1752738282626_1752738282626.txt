   const RequirementModal = ({ onAdd, onClose }) => {
     const [type, setType] = useState('age');
     const [value, setValue] = useState('');
     const [customLabel, setCustomLabel] = useState('');
     
     const requirementTypes = {
       age: { label: 'Minimum Age', unit: 'years', valueType: 'number' },
       tenure: { label: 'Employment Duration', unit: 'days', valueType: 'number' },
       license: { label: 'License/Certification', options: ['drivers', 'forklift', 'hazmat', 'food-handler'] },
       certification: { label: 'Special Certification', custom: true },
       approval: { label: 'Manager Approval', noValue: true },
       training: { label: 'External Training', custom: true },
       physical: { label: 'Physical Requirement', custom: true }
     };
     
     const handleAdd = () => {
       // Validation
       if (!requirementTypes[type].noValue && !value) {
         alert('Please enter a value');
         return;
       }
       
       if (requirementTypes[type].custom && !customLabel) {
         alert('Please enter a description');
         return;
       }
       
       // Generate label based on type
       let label = '';
       
       switch (type) {
         case 'age':
           label = `Must be ${value} years or older`;
           break;
         case 'tenure':
           const days = parseInt(value);
           if (days >= 365) {
             const years = Math.floor(days / 365);
             label = `Employed for at least ${years} year${years > 1 ? 's' : ''}`;
           } else if (days >= 30) {
             const months = Math.floor(days / 30);
             label = `Employed for at least ${months} month${months > 1 ? 's' : ''}`;
           } else {
             label = `Employed for at least ${days} days`;
           }
           break;
         case 'license':
           const licenseLabels = {
             'drivers': "Valid driver's license",
             'forklift': 'Forklift operator certification',
             'hazmat': 'HAZMAT certification',
             'food-handler': "Food handler's permit"
           };
           label = licenseLabels[value] || `${value} license required`;
           break;
         case 'approval':
           label = 'Requires manager approval';
           break;
         case 'certification':
         case 'training':
         case 'physical':
           label = customLabel;
           break;
         default:
           label = customLabel || `${requirementTypes[type].label} required`;
       }
       
       // Create requirement object
       const requirement = {
         type,
         value: requirementTypes[type].noValue ? null : value,
         label
       };
       
       console.log('Adding requirement:', requirement); // Debug log
       
       // Call the onAdd callback
       onAdd(requirement);
     };
     
     return (
       <div className="modal-overlay">
         <div className="modal-content small-modal">
           <h3>Add Course Requirement</h3>
           
           <label>
             Requirement Type
             <select 
               value={type} 
               onChange={(e) => {
                 setType(e.target.value);
                 setValue(''); // Reset value when type changes
                 setCustomLabel('');
               }}
             >
               {Object.entries(requirementTypes).map(([key, config]) => (
                 <option key={key} value={key}>{config.label}</option>
               ))}
             </select>
           </label>
           
           {!requirementTypes[type].noValue && (
             <>
               {requirementTypes[type].options ? (
                 <label>
                   Select {requirementTypes[type].label}
                   <select 
                     value={value} 
                     onChange={(e) => setValue(e.target.value)}
                     required
                   >
                     <option value="">Choose...</option>
                     {requirementTypes[type].options.map(opt => (
                       <option key={opt} value={opt}>
                         {opt.charAt(0).toUpperCase() + opt.slice(1).replace('-', ' ')}
                       </option>
                     ))}
                   </select>
                 </label>
               ) : requirementTypes[type].custom ? (
                 <label>
                   Requirement Description
                   <input
                     type="text"
                     value={customLabel}
                     onChange={(e) => setCustomLabel(e.target.value)}
                     placeholder="e.g., Must be able to lift 50 lbs"
                     required
                   />
                 </label>
               ) : (
                 <label>
                   {requirementTypes[type].label} ({requirementTypes[type].unit})
                   <input
                     type="number"
                     value={value}
                     onChange={(e) => setValue(e.target.value)}
                     placeholder={`Enter ${requirementTypes[type].unit}`}
                     min="0"
                     required
                   />
                 </label>
               )}
             </>
           )}
           
           <div className="modal-actions">
             <button 
               onClick={handleAdd} 
               className="btn-primary"
               type="button"
             >
               Add Requirement
             </button>
             <button 
               onClick={onClose} 
               className="btn-cancel"
               type="button"
             >
               Cancel
             </button>
           </div>
         </div>
       </div>
     );
   };
