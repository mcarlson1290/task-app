## Add the Frequency Label Helper Function

```javascript
const getFrequencyLabel = (frequency) => {
  switch (frequency) {
    case 'daily':
      return 'WEEKLY';
    case 'biweekly':
      return 'BI-WEEKLY';
    case 'monthly': 
      return 'MONTHLY';
    case 'quarterly':
      return 'QUARTERLY';
    default:
      return frequency.toUpperCase();
  }
};
```

## Option 4: Include Period Information for Bi-Weekly

For bi-weekly tasks, show which period it covers:

```javascript
const getFrequencyLabel = (frequency, task) => {
  switch (frequency) {
    case 'daily':
      return 'WEEKLY';
    case 'biweekly':
      // If task has periodLabel, use it
      return task.periodLabel ? 
        `BI-WEEKLY (${task.periodLabel})` : 
        'BI-WEEKLY';
    case 'monthly': 
      return 'MONTHLY';
    case 'quarterly':
      return task.periodLabel || 'QUARTERLY';
    default:
      return frequency.toUpperCase();
  }
};
```
