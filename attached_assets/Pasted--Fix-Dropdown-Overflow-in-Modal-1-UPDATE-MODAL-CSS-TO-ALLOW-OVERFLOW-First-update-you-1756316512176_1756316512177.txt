## 1. UPDATE MODAL CSS TO ALLOW OVERFLOW

First, update your modal CSS to allow the dropdown to extend outside:

```css
/* Fix modal overflow for dropdowns */
.modal-content {
  /* Remove overflow hidden if it exists */
  overflow: visible !important;
  position: relative;
  z-index: 1000;
}

/* If your modal body has scroll, only apply to content area */
.modal-body {
  /* Allow vertical scroll but visible horizontal */
  overflow-y: auto;
  overflow-x: visible;
  max-height: calc(90vh - 200px); /* Adjust based on your header/footer */
}

/* Ensure modal backdrop doesn't clip content */
.modal-overlay {
  overflow: visible;
}
```

## 2. USE A PORTAL FOR THE DROPDOWN (BETTER SOLUTION)

Create a dropdown that renders outside the modal using React Portal:

```javascript
// Update SearchableAssignmentSelect.js to use Portal
import React, { useState, useRef, useEffect } from 'react';
import { createPortal } from 'react-dom';

const SearchableAssignmentSelect = ({ 
  value, 
  onChange, 
  assignmentOptions, 
  placeholder = "Search or select assignment..." 
}) => {
  const [isOpen, setIsOpen] = useState(false);
  const [searchQuery, setSearchQuery] = useState('');
  const [highlightedIndex, setHighlightedIndex] = useState(0);
  const [dropdownPosition, setDropdownPosition] = useState({ top: 0, left: 0, width: 0 });
  const selectRef = useRef(null);
  const dropdownRef = useRef(null);
  const inputRef = useRef(null);
  
  // Calculate dropdown position when opening
  useEffect(() => {
    if (isOpen && selectRef.current) {
      const rect = selectRef.current.getBoundingClientRect();
      const spaceBelow = window.innerHeight - rect.bottom;
      const spaceAbove = rect.top;
      const dropdownHeight = 400; // max height of dropdown
      
      // Determine if dropdown should appear above or below
      const shouldFlip = spaceBelow < dropdownHeight && spaceAbove > spaceBelow;
      
      setDropdownPosition({
        top: shouldFlip ? rect.top - dropdownHeight : rect.bottom + 4,
        left: rect.left,
        width: rect.width,
        flip: shouldFlip
      });
    }
  }, [isOpen]);
  
  // Close dropdown when clicking outside
  useEffect(() => {
    const handleClickOutside = (event) => {
      if (
        selectRef.current && 
        !selectRef.current.contains(event.target) &&
        dropdownRef.current &&
        !dropdownRef.current.contains(event.target)
      ) {
        setIsOpen(false);
      }
    };
    
    if (isOpen) {
      document.addEventListener('mousedown', handleClickOutside);
      return () => document.removeEventListener('mousedown', handleClickOutside);
    }
  }, [isOpen]);
  
  // Get all options in a flat list (same as before)
  const getAllOptions = () => {
    const options = [];
    
    options.push({
      value: '',
      label: 'No default assignment',
      type: 'none'
    });
    
    assignmentOptions.special?.forEach(opt => {
      options.push({
        ...opt,
        searchText: opt.label.toLowerCase()
      });
    });
    
    assignmentOptions.roles?.forEach(opt => {
      options.push({
        ...opt,
        searchText: opt.label.toLowerCase()
      });
    });
    
    assignmentOptions.users?.forEach(opt => {
      options.push({
        ...opt,
        searchText: `${opt.label} ${opt.roles?.join(' ')}`.toLowerCase()
      });
    });
    
    return options;
  };
  
  // Filter options based on search
  const filteredOptions = getAllOptions().filter(option => 
    !searchQuery || option.searchText?.includes(searchQuery.toLowerCase())
  );
  
  // Get display text for selected value
  const getDisplayText = () => {
    if (!value) return 'No default assignment';
    
    const allOptions = getAllOptions();
    const selected = allOptions.find(opt => opt.value === value);
    
    if (selected) {
      if (selected.staffCount !== undefined) {
        return `${selected.label} (${selected.staffCount} ${selected.staffCount === 1 ? 'person' : 'people'})`;
      }
      return selected.label;
    }
    
    return value;
  };
  
  // Handle keyboard navigation (same as before)
  const handleKeyDown = (e) => {
    if (!isOpen) {
      if (e.key === 'Enter' || e.key === ' ' || e.key === 'ArrowDown') {
        setIsOpen(true);
        e.preventDefault();
      }
      return;
    }
    
    switch (e.key) {
      case 'ArrowDown':
        e.preventDefault();
        setHighlightedIndex(prev => 
          prev < filteredOptions.length - 1 ? prev + 1 : prev
        );
        break;
        
      case 'ArrowUp':
        e.preventDefault();
        setHighlightedIndex(prev => prev > 0 ? prev - 1 : 0);
        break;
        
      case 'Enter':
        e.preventDefault();
        if (filteredOptions[highlightedIndex]) {
          selectOption(filteredOptions[highlightedIndex]);
        }
        break;
        
      case 'Escape':
        setIsOpen(false);
        setSearchQuery('');
        break;
    }
  };
  
  const selectOption = (option) => {
    onChange(option.value);
    setIsOpen(false);
    setSearchQuery('');
    setHighlightedIndex(0);
  };
  
  useEffect(() => {
    setHighlightedIndex(0);
  }, [searchQuery]);
  
  // Render dropdown using portal
  const renderDropdown = () => {
    if (!isOpen) return null;
    
    return createPortal(
      <div 
        ref={dropdownRef}
        className={`select-dropdown-portal ${dropdownPosition.flip ? 'flip' : ''}`}
        style={{
          position: 'fixed',
          top: `${dropdownPosition.top}px`,
          left: `${dropdownPosition.left}px`,
          width: `${dropdownPosition.width}px`,
          zIndex: 9999
        }}
      >
        <div className="search-box">
          <input
            ref={inputRef}
            type="text"
            className="search-input"
            placeholder="Type to search..."
            value={searchQuery}
            onChange={(e) => setSearchQuery(e.target.value)}
            onKeyDown={handleKeyDown}
            autoFocus
          />
        </div>
        
        <div className="options-list">
          {filteredOptions.length === 0 ? (
            <div className="no-results">No matches found</div>
          ) : (
            <>
              {filteredOptions.filter(opt => opt.type === 'none').map((option, idx) => (
                <div
                  key={option.value}
                  className={`option ${highlightedIndex === idx ? 'highlighted' : ''}`}
                  onClick={() => selectOption(option)}
                >
                  {option.label}
                </div>
              ))}
              
              {filteredOptions.filter(opt => opt.type === 'special').length > 0 && (
                <div className="option-group-header">All Staff</div>
              )}
              {filteredOptions.filter(opt => opt.type === 'special').map((option) => {
                const actualIdx = filteredOptions.indexOf(option);
                return (
                  <div
                    key={option.value}
                    className={`option ${highlightedIndex === actualIdx ? 'highlighted' : ''}`}
                    onClick={() => selectOption(option)}
                  >
                    {option.label} ({option.staffCount} people)
                  </div>
                );
              })}
              
              {filteredOptions.filter(opt => opt.type === 'role').length > 0 && (
                <div className="option-group-header">Roles</div>
              )}
              {filteredOptions.filter(opt => opt.type === 'role').map((option) => {
                const actualIdx = filteredOptions.indexOf(option);
                return (
                  <div
                    key={option.value}
                    className={`option ${highlightedIndex === actualIdx ? 'highlighted' : ''}`}
                    onClick={() => selectOption(option)}
                  >
                    {option.label} ({option.staffCount} {option.staffCount === 1 ? 'person' : 'people'})
                  </div>
                );
              })}
              
              {filteredOptions.filter(opt => opt.type === 'user').length > 0 && (
                <div className="option-group-header">People</div>
              )}
              {filteredOptions.filter(opt => opt.type === 'user').map((option) => {
                const actualIdx = filteredOptions.indexOf(option);
                return (
                  <div
                    key={option.value}
                    className={`option ${highlightedIndex === actualIdx ? 'highlighted' : ''}`}
                    onClick={() => selectOption(option)}
                  >
                    {option.label}
                    {option.roles && option.roles.length > 0 && (
                      <span className="option-roles"> - {option.roles.join(', ')}</span>
                    )}
                  </div>
                );
              })}
            </>
          )}
        </div>
      </div>,
      document.body
    );
  };
  
  return (
    <div className="searchable-select" ref={selectRef}>
      <div 
        className="select-input"
        onClick={() => {
          setIsOpen(!isOpen);
          if (!isOpen) {
            setTimeout(() => inputRef.current?.focus(), 0);
          }
        }}
      >
        <div className="select-value">
          {getDisplayText()}
        </div>
        <div className="select-arrow">â–¼</div>
      </div>
      
      {renderDropdown()}
    </div>
  );
};

export default SearchableAssignmentSelect;
```

## 4. SIMPLER ALTERNATIVE - USE NATIVE SEARCHABLE

If the portal solution is too complex, here's a simpler approach using native HTML5 datalist:

```javascript
// Simpler searchable select using datalist
const SimpleSearchableSelect = ({ value, onChange, assignmentOptions }) => {
  const [inputValue, setInputValue] = useState('');
  const inputId = useRef(`select-${Date.now()}`).current;
  
  // Get all options
  const allOptions = [
    { value: '', label: 'No default assignment' },
    ...assignmentOptions.special.map(opt => ({
      value: opt.value,
      label: `${opt.label} (${opt.staffCount} people)`
    })),
    ...assignmentOptions.roles.map(opt => ({
      value: opt.value,
      label: `${opt.label} (${opt.staffCount} people)`
    })),
    ...assignmentOptions.users.map(opt => ({
      value: opt.value,
      label: `${opt.label} - ${opt.roles.join(', ')}`
    }))
  ];
  
  return (
    <div className="simple-searchable-select">
      <input
        list={inputId}
        value={inputValue}
        onChange={(e) => {
          const selected = allOptions.find(opt => opt.label === e.target.value);
          if (selected) {
            onChange(selected.value);
            setInputValue(selected.label);
          } else {
            setInputValue(e.target.value);
          }
        }}
        placeholder="Type to search assignments..."
        className="form-input"
      />
      <datalist id={inputId}>
        {allOptions.map(opt => (
          <option key={opt.value} value={opt.label} />
        ))}
      </datalist>
    </div>
  );
};
```

## The portal solution (Option 2) is the best approach because:
- Dropdown renders outside the modal boundaries
- Automatically positions above/below based on available space
- Works with any modal or container
- Maintains proper z-index layering

The key is using React Portal to render the dropdown at the document body level, completely outside the modal's DOM hierarchy.
