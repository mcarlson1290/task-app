### 2. COURSE CARD COMPONENT:

```typescript
const CourseCard: React.FC<CourseCardProps> = ({
  icon,
  title,
  description,
  sections,
  duration,
  status,
  completedDate,
  awards,
  requirementsMet,
  progress
}) => {
  const getStatusBadge = () => {
    switch (status) {
      case 'completed':
        return <span className="status-badge completed">‚úÖ Completed</span>;
      case 'in-progress':
        return <span className="status-badge in-progress">üîÑ In Progress</span>;
      case 'locked':
        return <span className="status-badge locked">üîí Locked</span>;
      default:
        return <span className="status-badge not-started">üìã Not Started</span>;
    }
  };
  
  return (
    <div className={`course-card ${status}`}>
      <div className="course-header">
        <div className="course-icon">{icon}</div>
        {getStatusBadge()}
      </div>
      
      <h3 className="course-title">{title}</h3>
      <p className="course-description">{description}</p>
      
      <div className="course-meta">
        <span className="meta-item">
          üìö {sections} sections
        </span>
        <span className="meta-item">
          ‚è±Ô∏è {duration}
        </span>
      </div>
      
      {awards && (
        <div className="course-awards">
          <span className="award-label">üèÜ Awards:</span>
          {awards.map((award, i) => (
            <span key={i} className="award-badge">{award}</span>
          ))}
        </div>
      )}
      
      {status === 'completed' && completedDate && (
        <div className="completion-info">
          <span className="completion-date">
            Completed on {completedDate}
          </span>
        </div>
      )}
      
      {status === 'in-progress' && progress && (
        <div className="course-progress">
          <div className="progress-bar-small">
            <div className="progress-fill" style={{ width: `${progress}%` }} />
          </div>
          <span className="progress-text">{progress}% complete</span>
        </div>
      )}
      
      {requirementsMet && status === 'not-started' && (
        <div className="requirements-met">
          ‚úÖ All Requirements Met
        </div>
      )}
      
      <div className="course-actions">
        {status === 'completed' ? (
          <button className="btn-course completed">
            ‚úÖ Completed
          </button>
        ) : status === 'in-progress' ? (
          <button className="btn-course resume">
            Resume Course
          </button>
        ) : status === 'locked' ? (
          <button className="btn-course locked" disabled>
            üîí Locked
          </button>
        ) : (
          <button className="btn-course start">
            Start Course
          </button>
        )}
      </div>
    </div>
  );
};
```
