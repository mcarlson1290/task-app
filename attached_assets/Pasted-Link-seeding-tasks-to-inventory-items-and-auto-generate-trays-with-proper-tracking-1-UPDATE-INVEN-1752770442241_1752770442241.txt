UPDATE TRAY GENERATION TO USE INVENTORY PRODUCT CODES:
javascriptconst generateTrayFromInventory = (seedInventoryItem, location = 'K') => {
  const date = new Date();
  const dateStr = `${(date.getMonth() + 1).toString().padStart(2, '0')}${date.getDate().toString().padStart(2, '0')}${date.getFullYear().toString().slice(-2)}`;
  
  // Determine category code based on seed type
  let categoryCode = 'XX';
  if (seedInventoryItem.name.includes('Microgreen')) {
    categoryCode = 'MG';
  } else if (seedInventoryItem.category === 'Seeds') {
    categoryCode = 'LG'; // Default for regular seeds
  }
  
  // Use product code from inventory
  const productCode = seedInventoryItem.productCode;
  
  // Generate tray identifier
  const trayNum = Math.floor(Math.random() * 9) + 1;
  const section = String.fromCharCode(65 + Math.floor(Math.random() * 4)); // A-D
  
  return `${location}${dateStr}-${categoryCode}-${productCode}-${trayNum}${section}`;
};

CREATE SEEDING TASK COMPLETION HANDLER WITH INVENTORY DEDUCTION:
javascriptconst handleSeedingTaskComplete = async (taskId, checklistData) => {
  const task = tasks.find(t => t.id === taskId);
  
  if (task && checklistData.seedInventoryId) {
    // Get seed inventory item
    const seedItem = inventoryItems.find(item => item.id === checklistData.seedInventoryId);
    
    if (seedItem) {
      const trayCount = checklistData.trayCount || 1;
      const seedsUsedOz = seedItem.ozPerTray * trayCount;
      const seedsUsedGrams = seedsUsedOz * 28.35; // Convert oz to grams
      
      // Check if enough inventory
      if (seedItem.quantity < seedsUsedGrams) {
        alert(`Not enough ${seedItem.name}! Need ${seedsUsedGrams}g but only have ${seedItem.quantity}g`);
        return;
      }
      
      // Generate trays with proper product code
      const generatedTrays = [];
      for (let i = 0; i < trayCount; i++) {
        const newTray = {
          id: generateTrayFromInventory(seedItem),
          cropType: seedItem.name.replace(' Seeds', ''),
          cropId: seedItem.cropId,
          productCode: seedItem.productCode,
          seedInventoryId: seedItem.id,
          seedLot: seedItem.lotNumber || 'Unknown', // If tracking lots
          datePlanted: new Date().toISOString().split('T')[0],
          assignedSystem: checklistData.systemLocation,
          estimatedHarvestDate: calculateHarvestDate(seedItem.cropId),
          status: 'growing',
          actualYield: null,
          expectedYield: crops.find(c => c.id === seedItem.cropId)?.expectedYieldPerTray,
          seedsUsedOz: seedItem.ozPerTray,
          notes: `Seeded by ${currentUser.name}`,
          createdByTask: task.id
        };
        
        generatedTrays.push(newTray);
      }
      
      // Update inventory
      updateInventoryItem(seedItem.id, {
        quantity: seedItem.quantity - seedsUsedGrams
      });
      
      // Add trays to production
      generatedTrays.forEach(tray => {
        TrayService.addTray(tray);
      });
      
      // Complete the task
      completeTask(taskId, {
        notes: checklistData.notes,
        traysGenerated: generatedTrays.map(t => t.id),
        seedsUsed: { 
          itemId: seedItem.id, 
          amount: seedsUsedGrams,
          unit: 'grams'
        }
      });
      
      // Show success
      alert(`âœ… Created ${trayCount} tray(s): ${generatedTrays.map(t => t.id).join(', ')}\nðŸ“‰ Used ${seedsUsedGrams}g of ${seedItem.name}`);
    }
  }
};
